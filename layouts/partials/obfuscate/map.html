{{/* obfuscate/map.html — deterministic class-name obfuscation mapping. Uses site.Params.obfuscateSeed + sha256 to generate stable 6-char aliases. */}} {{ $seed := site.Params.obfuscateSeed | default (site.Title | sha256) }} {{ $data := hugo.Data.obfuscate }} {{ $map := dict }} {{/* Protected classes: keep original name */}} {{ range $data.protected }} {{ $map = merge $map (dict . .) }} {{ end }} {{/* Obfuscated classes: deterministic short alias from sha256 */}} {{ range $data.classes }} {{ $clean := replace . "\\" "" }} {{ $hash := sha256 (printf "%s:%s" $seed $clean) }} {{ $short := substr $hash 0 6 }} {{ $map = merge $map (dict $clean $short) }} {{ end }} {{ return $map }}
